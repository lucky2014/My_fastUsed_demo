### 1、采取对象的方式写函数，在需要的地方调用，对象名必须为app，如：

	var app = {
		cardId: "123",
		//页面加载进来执行的函数
		init: function(){
			alert(0);
		}
	};

	//调用
	app.init();

### 2、上线的项目，一定要把console等调试的代码去掉。如：

	var app = {
		cardId: "123",
		//页面加载进来执行的函数
		init: function(msg){
			//console.log(msg);
			//alert(0);
		}
	};

	//调用
	app.init();

### 3、每个函数里必须缓存变量，名字为me。如：

	var app = {
		cardId: "123",
		//页面加载进来执行的函数
		init: function(msg){
			var me = this;
			//console.log(msg);
			alert(me.cardId);
		}
	};

	//调用
	app.init();

### 4、每个页面的页面逻辑函数必须是init。如：

	var app = {
		cardId: "123",
		//页面加载进来执行的函数
		init: function(msg){
			var me = this;
			//console.log(msg);
			alert(me.cardId);
		}
	};

	//调用
	app.init();
